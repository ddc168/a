# 第1题-两数之和
## 解法1：


    var twoSum = function(nums, target) {
        let x = 0, y = 0
        for(a in nums){
            b = nums.indexOf(target - nums[a])
            if(b >= 0 && a != b){
                if( a < b){
                    x = a
                    y = b
                } else {
                    x = b
                    y = a
                }
            }
            console.log(a, b, x, y)
        }
        return [x, y]
    }

# 第7题-整数反转
## 解法1：
    toString()  整数转换为字符串
    split("")   字符串拆分为字符数组
    reverse()   数组反转
    join("")    字符数组拼接为字符串
    parseInt()  字符串转换为整数
    判断正负数   负数要加负号
    判断是否超出Math.pow(2, 31)的范围

    var reverse = function(x) {
        x1 = parseInt(x.toString().split("").reverse().join(""))
        if(x >= 0 ){
            if(x1 < Math.pow(2, 31)){
                return x1
            } else {
                return 0
            }
        } else {
            if(-x1 > -Math.pow(2, 31)+1){
                return -x1
            } else {
                return 0
            }
        }
    };
